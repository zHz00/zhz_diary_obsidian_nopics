  Как нормальные люди пишут код, если им надо вызвать несколько функций подряд?   
 
```
  
void a();  
void b();  
void c();  
void d();  
  
int main()  
{  
	a();  
	b();  
	c();  
	d();  
	return 0;  
}  

```
   
   
 Как пишут все остальные?   
 
```
  
void a();  
void b();  
void c();  
void d();  
  
int main()  
{  
	a();  
	return 0;  
}  
  
void a()  
{  
	// тут какие-то действия  
	b();  
}  
  
void b()  
{  
	// тут какие-то действия  
	c();  
}  
  
void c()  
{  
	// тут какие-то действия  
	d();  
}  
  

```
